\begin{frame}[fragile]
\head{The PLTC Section}

\vfill

\begin{itemize}
\item Conducts research in programming language technology and the theory of computation.

\vfill

\item Research involves topics in the intersection of

\vfill
\begin{center}

\textbf{\large programming language theory}

(e.g., type systems, compilers, DSLs, parallelisation, and formal verification)

\vfill

and

\vfill

\textbf{\large applications}

(e.g., security and privacy, systems,
distributed ledgers, fintech, quantum, ML, LLMs)
\end{center}
\end{itemize}

\end{frame}

\begin{frame}[fragile]

  \head{Programming Languages as Research Objects!}

  \vfill

  \begin{itemize}
  \item How do we implement and design good programming languages?  What
    does ``good'' mean? \vfill

  \item What does correctness mean? \vfill

  \item What does it take for software to be secure? \vfill

  \item How can we organize parallel computations optimally? \vfill

  \item What does ``optimal'' mean? \vfill

  \item How are ML and LLMs implemented? \vfill

  \item ...  \vfill

  \end{itemize}

\end{frame}
